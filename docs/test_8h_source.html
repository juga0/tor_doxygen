95 /*
96  * These macros are helpful for streamlining the authorship of several test
97  * cases that use mocks.
98  *
99  * The pattern is as follows.
100  * * Declare a top level namespace:
101  * #define NS_MODULE foo
102  *
103  * * For each test case you want to write, create a new submodule in the
104  * namespace. All mocks and other information should belong to a single
105  * submodule to avoid interference with other test cases.
106  * You can simply name the submodule after the function in the module you
107  * are testing:
108  * #define NS_SUBMODULE some_function
109  * or, if you're wanting to write several tests against the same function,
110  * ie., you are testing an aspect of that function, you can use:
111  * #define NS_SUBMODULE ASPECT(some_function, behavior)
112  *
113  * * Declare all the mocks you will use. The NS_DECL macro serves to declare
114  * the mock in the current namespace (defined by NS_MODULE and NS_SUBMODULE).
115  * It behaves like MOCK_DECL:
116  * NS_DECL(int, dependent_function, (void *));
117  * Here, dependent_function must be declared and implemented with the
118  * MOCK_DECL and MOCK_IMPL macros. The NS_DECL macro also defines an integer
119  * global for use for tracking how many times a mock was called, and can be
120  * accessed by CALLED(mock_name). For example, you might put
121  * CALLED(dependent_function)++;
122  * in your mock body.
123  *
124  * * Define a function called NS(main) that will contain the body of the
125  * test case. The NS macro can be used to reference a name in the current
126  * namespace.
127  *
128  * * In NS(main), indicate that a mock function in the current namespace,
129  * declared with NS_DECL is to override that in the global namespace,
130  * with the NS_MOCK macro:
131  * NS_MOCK(dependent_function)
132  * Unmock with:
133  * NS_UNMOCK(dependent_function)
134  *
135  * * Define the mocks with the NS macro, eg.,
136  * int
137  * NS(dependent_function)(void *)
138  * {
139  * CALLED(dependent_function)++;
140  * }
141  *
142  * * In the struct testcase_t array, you can use the TEST_CASE and
143  * TEST_CASE_ASPECT macros to define the cases without having to do so
144  * explicitly nor without having to reset NS_SUBMODULE, eg.,
145  * struct testcase_t foo_tests[] = {
146  * TEST_CASE_ASPECT(some_function, behavior),
147  * ...
148  * END_OF_TESTCASES
149  * which will define a test case named "some_function__behavior".
150  */
