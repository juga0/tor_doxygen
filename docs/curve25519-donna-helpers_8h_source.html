47  * z^(p - 2) = z(2^255 - 21)
48  */
49 static void
50 curve25519_recip(bignum25519 out, const bignum25519 z) {
51  bignum25519 ALIGN(16) a,t0,b;
52 
53  /* 2 */ curve25519_square_times(a, z, 1); /* a = 2 */
54  /* 8 */ curve25519_square_times(t0, a, 2);
55  /* 9 */ curve25519_mul_noinline(b, t0, z); /* b = 9 */
56  /* 11 */ curve25519_mul_noinline(a, b, a); /* a = 11 */
57  /* 22 */ curve25519_square_times(t0, a, 1);
58  /* 2^5 - 2^0 = 31 */ curve25519_mul_noinline(b, t0, b);
59  /* 2^250 - 2^0 */ curve25519_pow_two5mtwo0_two250mtwo0(b);
60  /* 2^255 - 2^5 */ curve25519_square_times(b, b, 5);
61  /* 2^255 - 21 */ curve25519_mul_noinline(out, b, a);
62 }
