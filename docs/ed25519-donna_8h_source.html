76 static int
77 ed25519_verify(const unsigned char *x, const unsigned char *y, size_t len) {
78  size_t differentbits = 0;
79  while (len--)
80  differentbits |= (*x++ ^ *y++);
81  /*coverity[overflow]*/
82  return (int) (1 & ((differentbits - 1) >> 8));
83 }
