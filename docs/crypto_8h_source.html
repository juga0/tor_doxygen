23 /*
24  Macro to create an arbitrary OpenSSL version number as used by
25  OPENSSL_VERSION_NUMBER or SSLeay(), since the actual numbers are a bit hard
26  to read.
27 
28  Don't use this directly, instead use one of the other OPENSSL_V macros
29  below.
30 
31  The format is: 4 bits major, 8 bits minor, 8 bits fix, 8 bits patch, 4 bit
32  status.
33  */
34 #define OPENSSL_VER(a,b,c,d,e) \
35  (((a)<<28) | \
36  ((b)<<20) | \
37  ((c)<<12) | \
38  ((d)<< 4) | \
39  (e))
40 
42 #define OPENSSL_V(a,b,c,d) \
43  OPENSSL_VER((a),(b),(c),(d)-'a'+1,0xf)
44 
47 #define OPENSSL_V_NOPATCH(a,b,c) \
48  OPENSSL_VER((a),(b),(c),0,0xf)
49 
52 #define OPENSSL_V_SERIES(a,b,c) \
53  OPENSSL_VER((a),(b),(c),0,0)
