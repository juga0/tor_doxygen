113 static inline int
114 download_status_is_ready(download_status_t *dls, time_t now,
115  int max_failures)
116 {
117  if (dls->backoff == DL_SCHED_DETERMINISTIC) {
118  /* Deterministic schedules can hit an endpoint; exponential backoff
119  * schedules just wait longer and longer. */
120  int under_failure_limit = (dls->n_download_failures <= max_failures
121  && dls->n_download_attempts <= max_failures);
122  if (!under_failure_limit)
123  return 0;
124  }
125  return dls->next_attempt_at <= now;
126 }
